#title: 덧셈 함수
#desc: 두 수를 더하는 함수 작성
def add(a, b):
    return a + b
---
#title: 최대값 찾기
#desc: 리스트에서 최댓값을 반환
def find_max(nums):
    return max(nums)
---
#title: 짝수 홀수 판별
#desc: 숫자가 짝수인지 홀수인지 판별
def is_even(number):
    return number % 2 == 0
---
#title: 리스트 길이 구하기
#desc: 리스트에 포함된 요소의 개수 반환
def get_length(items):
    return len(items)
---
#title: 문자열 뒤집기
#desc: 주어진 문자열을 거꾸로 뒤집기
def reverse_string(s):
    return s[::-1]
---
#title: 리스트의 모든 요소 합산
#desc: 숫자 리스트의 모든 요소를 더한 값을 반환
def sum_list(numbers):
    return sum(numbers)
